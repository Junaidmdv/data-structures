

func DetectCycle(list *ListNode){
     slow:=list.Head 
     fast:=list.Head 

     for fast != nil && fast.Next != nil{
         fast=fast.Next.Next
         slow=slow.Next 

         if slow==fast{  // if the slow equal to the fast cycle detected
            break  
         }
     }


     slow=list.Head 

     for fast != slow{
        fast=fast.Next 
        slow=slow.Next 
     }

     for fast.Next != slow{
        fast=fast.Next
     }

     fast.Next=nil
}
